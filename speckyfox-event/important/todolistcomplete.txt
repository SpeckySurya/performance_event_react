import { useState } from 'react';
import './App.css';

function App() {
  const [data, setdata] = useState("");
  const [item, setitems] = useState([]);

  const listofitem = () => {
    setitems((olditem) => {
      return [...olditem, data];
    });
    setdata("");
  };

  return (
    <div className="Appctr">
      <div className='childctr'>
        <br />
        <h1>TODo List</h1>
        <br />
        <input type='text' placeholder="Add a value" value={data} onChange={(e) => setdata(e.target.value)} />
        <button onClick={listofitem}>+</button>
        <ol>
        
          {
            item.map((val) => { return <li>{val}</li> })
          }
        </ol>
      </div>

    </div>
  );
}

export default App;
